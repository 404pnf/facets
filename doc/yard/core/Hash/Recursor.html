def method_missing(op, &yld)
  yld = yld    || lambda{ |k,v| [k,v] }  # ? to_enum
  rec = @block || yld #lambda{ |k,v| [k,v] }
  @enum.__send__(op) do |k,v|
    case v
    when String # b/c of 1.8
      yld.call(k,v)
    when *@types
      res = v.recursively(*@types, &@block).__send__(op,&yld)
      rec.call(k, res)
    else
      yld.call(k,v)
    end
  end
end
