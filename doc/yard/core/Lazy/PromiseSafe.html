def __synchronize__ 
  current = Thread.current

  Thread.critical = true
  unless @computation
    # fast path for evaluated thunks
    Thread.critical = false
    yield
  else
    if @owner == current
      Thread.critical = false
      raise DivergenceError.new
    end
    while @owner # spinlock
      Thread.critical = false
      Thread.pass
      Thread.critical = true
    end
    @owner = current
    Thread.critical = false

    begin
      yield
    ensure
      @owner = nil
    end
  end
end
