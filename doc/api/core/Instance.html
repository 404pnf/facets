  end
end

f1.instance

to_h(at=false) 
# File lib/core/facets/instance.rb, line 62
def to_h(at=false)
  h = {}
  if at
    @delegate.instance_variables.each do |name|
      h[name] = @delegate.instance_variable_get(name)
    end
  else
    each do |key, value|
      h[key] = value
    end
  end
  h
end

Return instance variables with values as a hash.

class X
  def initialize(a,b)
    @a, @b = a, b
  end
end

x = X.new(1,2)

x.instance.to_h  #=> { :a=>1, :b=>2 }
