def initialize( &computation ) #:nodoc:
  result = nil
  exception = nil

  thread = Thread.new do
    begin
      result = computation.call( self )
    rescue Exception => exception
    end
  end

  super() do
    raise DivergenceError.new if Thread.current == thread
    thread.join
    raise exception if exception
    result
  end
end
