 31   # Allows the initialization of an OpenStruct with a setter block:
 32   #
 33   #   person = OpenStruct.new do |o|
 34   #     o.name    = 'John Smith'
 35   #     o.gender  = :M
 36   #     o.age     = 71
 37   #   end 
 38   #
 39   # You can still provide a hash for initialization purposes, and even combine
 40   # the two approaches if you wish.
 41   #
 42   #   person = OpenStruct.new(:name => 'John Smith', :age => 31) do |p|
 43   #     p.gender = :M 
 44   #   end
 45   #
 46   # Alternatively you can provide a default block:
 47   #
 48   #   stuff = OpenStruct.new{ |o,k| o[k] = [] }
 49   #   stuff.place << :a
 50   #   stuff.place << :b
 51   #   stuff.place #=> [:a, :b]
 52   #
 53   # A setter block versus a defualt block is determined by the arity of
 54   # the block. You can not provide both at the same time.
 55   #
 56   # CREDIT: Noah Gibbs, Gavin Sinclair
 57   #
 58   def initialize(hash=nil, &block)
 59     if block && block.arity==2
 60       @table = Hash.new(&block)
 61     else
 62       @table = {}
 63     end
 64     if hash
 65       for k,v in hash
 66         @table[k.to_sym] = v
 67         new_ostruct_member(k)
 68       end
 69     end
 70     if block && block.arity==1
 71       yield self
 72     end
 73   end

211   # Like to_ostruct but recusively objectifies all hash elements as well.
212   #
213   #     o = { 'a' => { 'b' => 1 } }.to_ostruct_recurse
214   #     o.a.b  #=> 1
215   #
216   # The +exclude+ parameter is used internally to prevent infinite
217   # recursion and is not intended to be utilized by the end-user.
218   # But for more advance use, if there is a particular subhash you
219   # would like to prevent from being converted to an OpoenStruct
220   # then include it in the +exclude+ hash referencing itself. Eg.
221   #
222   #     h = { 'a' => { 'b' => 1 } }
223   #     o = h.to_ostruct_recurse( { h['a'] => h['a'] } )
224   #     o.a['b']  #=> 1
225   #
226   def to_ostruct_recurse(exclude={})
227     return exclude[self] if exclude.key?( self )
228     o = exclude[self] = OpenStruct.new
229     h = self.dup
230     each_pair do |k,v|
231       h[k] = v.to_ostruct_recurse( exclude ) if v.respond_to?(:to_ostruct_recurse)
232     end
233     o.__update__(h)
234   end
