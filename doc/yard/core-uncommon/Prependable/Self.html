def method_added(meth)
  return if meth.to_s[-2,2] == ':-'
  alias_method "#{meth}:-", meth
  remove_method(meth)
  prepend_module.module_eval %{
    def #{meth}(*a,&b); __send__("#{meth}:-",*a,&b); end
  }
  include prepend_module
end
