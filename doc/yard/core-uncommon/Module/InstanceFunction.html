def singleton_method_added(meth)
  this = self
  ##module_eval %{
  ##  def #{meth}(*args)
  ##    #{self.name}.#{meth}(self,*args)
  ##  end
  ##}
  module_eval do
    define_method(meth) do |*args|
      this.__send__(meth, self, *args)
    end
  end
  super(meth)
end
