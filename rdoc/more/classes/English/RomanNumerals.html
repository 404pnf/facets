  def from_integer(integer)
    return nil if integer < 0 || integer > MAX

    r = integer  # remainder
    y = ''       # result

    NUMBERS.each do |n|
      while r >= n
        r -= n
        y += LOOKUP[n]
      end
      break if r <= 0
    end

    return y
  end

  def to_integer(roman_string)
    return nil unless roman_string.is_roman_numeral?

    l = nil  # last
    i = 0    # integer result

    c = roman_string.to_s.upcase.split(//).reverse

    c.each do |d|
      if v = TABLE[d]
        if l && l > v
          i -= v
        else
          i += v
        end
        l = v
      end
    end

    return i
  end
