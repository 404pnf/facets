def roman
  int = self
  #return nil if integer > ROMAN_MAX
  return "-#{(-int).roman}" if int < 0
  return "" if int == 0
  ROMAN_VALUES.each do |(i, v)|
    return(i + (int-v).roman) if v <= int 
  end
end
