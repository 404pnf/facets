def at_rand!( separator=// )
  #separator = self.class.patterns( separator )
  a = self.shatter( separator )
  w = []; a.each_with_index { |s,i| i % 2 == 0 ? w << s : w.last << s }
  i = Random.number(w.size)
  r = w.delete_at( i )
  self.replace( w.join('') )
  return r
end
