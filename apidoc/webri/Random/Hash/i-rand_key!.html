Delete a random key-value pair, returning the key.

  a = {:one =&gt; 1, :two =&gt; 2, :three =&gt; 3}
  a.pick_key!  #=&gt; :two
  a            #=&gt; {:one =&gt; 1, :three =&gt; 3}
