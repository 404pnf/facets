def random(max_length = 8, char_re = /[\w\d]/)
  raise ArgumentError.new('char_re must be a regular expression!') unless char_re.is_a?(Regexp)
  string = ""
  while string.length < max_length
      ch = Random.number(255).chr
      string << ch if ch =~ char_re
  end
  return string
end
