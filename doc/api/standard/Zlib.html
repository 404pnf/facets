Zlib

A convenient wrapper for the zlib standard library that allows compression/decompression of strings with gzip.

def self.compress(source)
  output = StringIO.new
  class << output
    def close; rewind; end
  end
  gz = GzipWriter.new(output)
  gz.write(source)
  gz.close
  output.string
end
