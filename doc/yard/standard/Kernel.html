def safe_memo(*args, &block)
  if args.empty?
    args = block.binding.eval('[self, __method__]')
  end
  $MEMO_MUTEX.synchronize do
    if $MEMO.key?(args)
      $MEMO[args]
    else
      $MEMO[args] = block.call
    end
  end
end
